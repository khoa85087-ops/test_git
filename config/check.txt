Write-Host "`n=== APP DANG TRUY CAP MANG ===" -ForegroundColor Cyan

# IP pho bien/an toan
$safeIPs = @(
    "127.",
    "192.168.",
    "172.217.",   # Google
    "142.251.",   # Google
    "157.240.",   # Facebook/Meta
    "104.16.",
    "104.17.",
    "104.18.",
    "104.19.",
    "104.20.",
    "104.21.",
    "104.22.",
    "104.26.",    # Cloudflare
    "1.1.1.1",
    "8.8.8.8"
)

Get-NetTCPConnection -State Established |
ForEach-Object {

    try {

        $proc = Get-Process -Id $_.OwningProcess -ErrorAction SilentlyContinue

        $safe = "NON"

        foreach ($ip in $safeIPs) {

            if ($_.RemoteAddress.StartsWith($ip)) {
                $safe = "SAFE"
            }
        }

        [PSCustomObject]@{
            App        = $proc.ProcessName
            PID        = $_.OwningProcess
            RemoteIP   = $_.RemoteAddress
            RemotePort = $_.RemotePort
            Status     = $safe
        }

    } catch {}

} | Sort-Object Status,App |
Format-Table -AutoSize


Write-Host "`n=== APP KHOI DONG CUNG WINDOWS ===" -ForegroundColor Yellow

Select-Object Name, Command |

# Lấy path thật của Look
$look = (Get-Process lookapp).Path

Write-Host "LOOK PATH:"
Write-Host $look
Write-Host ""

# Xóa rule cũ
Get-NetFirewallRule |
Where-Object DisplayName -like "LOOK_ONLY*" |
Remove-NetFirewallRule

# Block outbound
New-NetFirewallRule `
-DisplayName "LOOK_ONLY_OUT" `
-Direction Outbound `
-Program $look `
-Action Block `
-Profile Any

# Block inbound
New-NetFirewallRule `
-DisplayName "LOOK_ONLY_IN" `
-Direction Inbound `
-Program $look `
-Action Block `
-Profile Any

Write-Host ""
Write-Host "Look blocked."
