$exe = "C:\Program Files\BraveSoftware\Brave-Browser\Application\brave.exe"

New-NetFirewallRule `
    -DisplayName "Block brave Outbound" `
    -Direction Outbound `
    -Program $exe `
    -Action Block `
    -Profile Any

New-NetFirewallRule `
    -DisplayName "Block brave Inbound" `
    -Direction Inbound `
    -Program $exe `
    -Action Block `
    -Profile Any
